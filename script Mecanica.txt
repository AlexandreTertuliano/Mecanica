create table Clientes(
	cpf varchar(14) PRIMARY KEY, 
	Nome varchar(50) not null,
	Rg varchar(15) not null,
	Cell varchar(15) not null,
	Fixo varchar (10) not null,

	Estado varchar (20) not null,
	Cep varchar (20) not null,
	Bairro varchar (50) not null,
	Rua varchar (50) not null,
	Numero varchar (10) not null,
	Cidade varchar (20) not null,
	Apt_Casa float,

	Limite float,
	Bloquear float
);

create table Placa_Veiculos (
	placa varchar(10) PRIMARY KEY,
	cliente varchar (50) not null,
	Km varchar (50) not null,
	Modelo varchar (50) not null
);

create table Produtos(
	Cod_Barras varchar (15) PRIMARY KEY,
	Cod_Sistema varchar (6) not null,
	Descricao varchar (50) not null,
	Data_Inclusao varchar (10) not null,
	Fornecedor varchar (50) ,
	Quantidade varchar (10) not null,
	Preco_Custo varchar (10) not null,
	Preco_Venda varchar (10) not null
);




create table Funcionarios(
	cpf varchar(14) PRIMARY KEY, 
	Nome varchar(50) not null,
	Senha varchar(15) not null,
	
);

create table Boleto(
	Descricao varchar (50) not null,
	valor varchar (20) not null,
	Data_Vencimento varchar (10) not null,
	Parcelas varchar (5) not null,
	Soma_boletos varchar (10) not null
); 

delete from Produtos;
delete from Placa_Veiculos;
delete from clientes;
delete from Funcionarios;
delete from fornecedores;
delete from Boleto;
